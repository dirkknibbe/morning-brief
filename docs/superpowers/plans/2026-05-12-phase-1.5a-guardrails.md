# Phase 1.5a Guardrails (Tier 1) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Harden the existing ideas pipeline with deterministic boundaries — a status-transition state machine, an append-only audit log, a Mongo schema validator, and a system-wide kill switch (`/freeze` + per-stage enable flags).

**Architecture:** Pure logic modules (`status.ts`, `system-state.ts canRun`) backed by Mongo I/O modules (`audit.ts`, `system-state.ts CRUD`), wired into the existing `ideas-state.ts` and `extract-ideas.ts` call sites. The bash-level kill switch in `loop-triggers.sh` shells out to a new `bun run system-state check <stage>` subcommand to avoid a `jq` dependency. The Mongo `$jsonSchema` validator ships in `warn` mode initially; promotion to `strict` is a separate manual one-shot.

**Tech Stack:** Bun, TypeScript, MongoDB ^7.1.1, `bun:test`.

**Spec:** [docs/superpowers/specs/2026-05-12-phase-1.5-guardrails-design.md](../specs/2026-05-12-phase-1.5-guardrails-design.md). Phase 1.5a (Tier 1) only — Tier 2 factory prerequisites (lock, worktree guard, pre-push hook, deny list) ship in Phase 1.5b.

**Out of scope for Phase 1.5a (deferred to 1.5b):**
- `src/factory-lock.ts`
- `src/factory-guard.ts`
- `scripts/hooks/pre-push` + installer
- `.claude/settings.json` deny-list expansion

```ts
/**
 * status.ts — idea status type and state-machine transition rules.
 *
 * Pure module. No I/O, no Mongo. Imported by ideas-state.ts.
 */

export type Status =
  | "extracted"
  | "queued"
  | "building"
  | "built"
  | "parked"
  | "rejected"
  | "needs_human";

export const ALL_STATUSES: readonly Status[] = [
  "extracted",
  "queued",
  "building",
  "built",
  "parked",
  "rejected",
  "needs_human",
];

export const ALLOWED_TRANSITIONS: Record<Status, readonly Status[]> = {
  extracted:   ["queued", "rejected", "needs_human"],
  queued:      ["building", "extracted", "rejected", "parked"],
  building:    ["built", "parked", "needs_human", "queued"],
  built:       [],
  parked:      ["queued", "rejected"],
  rejected:    [],
  needs_human: ["queued", "rejected"],
};

export class IllegalTransitionError extends Error {
  constructor(
    public readonly from: Status,
    public readonly to: Status,
    public readonly allowed: readonly Status[],
  ) {
    super(`Illegal transition: ${from} → ${to}. Allowed from "${from}": [${allowed.join(", ")}]`);
    this.name = "IllegalTransitionError";
  }
}

export function isValidStatus(s: string): s is Status {
  return (ALL_STATUSES as readonly string[]).includes(s);
}

export function isValidTransition(from: Status, to: Status): boolean {
  return ALLOWED_TRANSITIONS[from].includes(to);
}

export function assertValidTransition(from: Status, to: Status): void {
  if (!isValidTransition(from, to)) {
    throw new IllegalTransitionError(from, to, ALLOWED_TRANSITIONS[from]);
  }
}
```

```ts
export async function applyUpsertOp(db: Db, op: UpsertOp, actor: string): Promise<void> {
  if (op.kind === "skip") return;
  if (op.kind === "insert") {

      await db.collection("ideas").insertOne(op.doc as any);
      await recordTransition(db, op.doc.slug, null, "extracted", actor);

  await db.collection("ideas").updateOne(
    { slug: op.slug },
    {
      $inc: { signal_strength: 1 },
      $addToSet: { sources: op.new_source as any },
```

Create `src/system-state.ts`:

```ts
/**
 * system-state.ts — system-wide kill switch and per-stage enable flags.
 *
 * Mongo I/O + CLI. Pure `canRun` exported for testing the gating decision.
 *
 * Library API (used by extract-ideas.ts and other triggers):
 *   isFrozen(db) → boolean
 *   isEnabled(db, stage) → boolean
 *   canRun(state, stage) → boolean   (pure)
 *   setFrozen(db, frozen, reason, actor)
 *   setEnabled(db, stage, enabled, actor)
 *   getState(db) → SystemState
 *
 * CLI:
 *   status                        — print current state as JSON
 *   check <stage>                 — exit 0 if stage may run, 1 otherwise (for shell gates)
 *   freeze [reason]               — set frozen=true
 *   unfreeze                      — set frozen=false
 *   enable  <stage>               — set <stage>_enabled=true
 *   disable <stage>               — set <stage>_enabled=false
 */

import { MongoClient, type Db } from "mongodb";

export type Stage = "extract" | "synthesize" | "triage" | "factory";

const STAGES: readonly Stage[] = ["extract", "synthesize", "triage", "factory"];

function isStage(s: string): s is Stage {
  return (STAGES as readonly string[]).includes(s);
}

export interface SystemState {
  frozen: boolean;
  extract_enabled: boolean;
  synthesize_enabled: boolean;
  triage_enabled: boolean;
  factory_enabled: boolean;
  freeze_reason: string | null;
  updated_at: Date;
  updated_by: string;
}

const SINGLETON_ID = "singleton";

const DEFAULT_STATE: Omit<SystemState, "updated_at" | "updated_by"> = {
  frozen: false,
  extract_enabled: true,
  synthesize_enabled: true,
  triage_enabled: true,
  factory_enabled: true,
  freeze_reason: null,
};

export function canRun(state: SystemState, stage: Stage): boolean {
  if (state.frozen) return false;
  switch (stage) {
    case "extract":    return state.extract_enabled;
    case "synthesize": return state.synthesize_enabled;
    case "triage":     return state.triage_enabled;
    case "factory":    return state.factory_enabled;
  }
}

export async function getState(db: Db): Promise<SystemState> {
  const doc = await db.collection("system_state").findOne({ _id: SINGLETON_ID as any });
  if (!doc) {
    throw new Error("system_state singleton missing — run `bun run init-db`");
  }
  return {
    frozen: doc.frozen,
    extract_enabled: doc.extract_enabled,
    synthesize_enabled: doc.synthesize_enabled,
    triage_enabled: doc.triage_enabled,
    factory_enabled: doc.factory_enabled,
    freeze_reason: doc.freeze_reason ?? null,
    updated_at: doc.updated_at,
    updated_by: doc.updated_by,
  };
}

export async function isFrozen(db: Db): Promise<boolean> {
  return (await getState(db)).frozen;
}

export async function isEnabled(db: Db, stage: Stage): Promise<boolean> {
  const state = await getState(db);
  return canRun(state, stage);
}

export async function setFrozen(
  db: Db,
  frozen: boolean,
  reason: string | null,
  actor: string,
): Promise<void> {
  await db.collection("system_state").updateOne(
    { _id: SINGLETON_ID as any },
    {
      $set: {
        frozen,
        freeze_reason: frozen ? reason : null,
        updated_at: new Date(),
        updated_by: actor,
      },
    },
  );
}

  db: Db,
  stage: Stage,
  enabled: boolean,

      $set: {
        [field]: enabled,
        updated_at: new Date(),
        updated_by: actor,
      },
    },
  );
}
```
